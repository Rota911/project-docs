# Config fájlok

Az LS Core különböző **konfigurációs fájlokat** használ a szerver beállításához. Ezek a fájlok a `core/configs/` mappában találhatók.

## Admin csoportok

### sv_adminGroups.lua

Az admin csoportok hierarchiájának definiálása.

```lua
-- core/configs/sv_adminGroups.lua
return {
    ['owner'] = {
        label = 'Tulajdonos',
        child = 'dev'
    },
    ['dev'] = {
        label = 'Fejlesztő',
        child = 'headadmin'
    },
    ['headadmin'] = {
        label = 'Főadmin',
        child = 'admin'
    },
    ['admin'] = {
        label = 'Admin',
        child = 'moderator'
    },
    ['moderator'] = {
        label = 'Moderátor',
        child = 'user'
    },
    ['user'] = {
        label = 'Felhasználó',
        child = nil
    }
}
```

### Struktúra

```lua
{
    ['csoportNev'] = {
        label = 'Megjelenítési név',  -- UI-ban megjelenő név
        child = 'alatteLevoCsoport'   -- Gyerek csoport (nil = nincs)
    }
}
```

**Hierarchia működése:**
- Minden csoport örökli az alatta lévő csoport jogosultságait
- `owner` a legmagasabb szint (minden jogosultsága van)
- `user` az alapértelmezett csoport (minden játékos)

### Új admin csoport hozzáadása

```lua
return {
    ['owner'] = {
        label = 'Tulajdonos',
        child = 'manager'  -- Új csoport beillesztése
    },
    ['manager'] = {
        label = 'Manager',
        child = 'dev'
    },
    ['dev'] = {
        label = 'Fejlesztő',
        child = 'headadmin'
    },
    -- ... többi csoport
}
```

**Eredmény:**
```
owner → manager → dev → headadmin → admin → moderator → user
```

## Spawn helyek

### sh_locations.lua

Spawn és loading koordináták definiálása.

```lua
-- core/configs/sh_locations.lua
return {
    Loading = vector3(117.47, -750.09, 45.75),
    Spawn = vector3(-269.94, -955.14, 31.22)
}
```

### Struktúra

```lua
{
    Loading = vector3(x, y, z),  -- Betöltési pozíció
    Spawn = vector3(x, y, z)     -- Spawn pozíció
}
```

**Használat:**
- `Loading`: Karakter kiválasztás közben itt spawnolja a játékost
- `Spawn`: Alapértelmezett spawn koordináta (ha nincs mentett pozíció)

### Koordináta lekérése in-game

```lua
-- Kliens oldalon (F8 konzolban)
local coords = GetEntityCoords(PlayerPedId())
print(coords)

-- Kimenet:
vector3(-269.94, -955.14, 31.22)
```

### Példa: Több spawn pont

```lua
-- sh_locations.lua
return {
    Loading = vector3(117.47, -750.09, 45.75),
    
    Spawn = vector3(-269.94, -955.14, 31.22),
    
    -- További spawn pontok (opcionális)
    SpawnPoints = {
        hospital = vector3(298.8, -584.69, 43.26),
        police = vector3(425.13, -979.55, 30.71),
        garage = vector3(-340.76, -874.03, 31.31),
        airport = vector3(-1037.74, -2738.52, 13.76)
    }
}
```

### Koordináták használata

```lua
local Locations = require('core.configs.sh_locations')

-- Loading pozíció
local loadingPos = Locations.Loading
SetEntityCoords(ped, loadingPos.x, loadingPos.y, loadingPos.z)

-- Spawn pozíció
local spawnPos = Locations.Spawn
TriggerEvent('playerSpawned', spawnPos)

-- Ha van mentett pozíció
local character = GetCharacter(source)
local savedCoords = character:GetCoords()

if savedCoords then
    character:SetCoords(savedCoords)
else
    -- Alapértelmezett spawn
    character:SetCoords(Locations.Spawn)
end
```

## Database config

### sv_database.lua

Adatbázis kapcsolat konfigurációja.

```lua
-- core/configs/sv_database.lua
return {
    host = '127.0.0.1',
    port = 3306,
    database = 'essentialmode',
    username = 'root',
    password = ''
}
```

### Struktúra

```lua
{
    host = 'string',      -- MySQL szerver címe
    port = number,        -- Port (alapértelmezett: 3306)
    database = 'string',  -- Adatbázis neve
    username = 'string',  -- Felhasználónév
    password = 'string'   -- Jelszó
}
```

**Biztonság:**
- Ne commitáld a jelszót git-be
- Használj környezeti változókat production környezetben
- Korlátozott jogosultságú felhasználó ajánlott

### Példa: Biztonságos konfig

```lua
-- sv_database.lua
return {
    host = GetConvar('db_host', '127.0.0.1'),
    port = GetConvarInt('db_port', 3306),
    database = GetConvar('db_name', 'essentialmode'),
    username = GetConvar('db_user', 'root'),
    password = GetConvar('db_pass', '')
}
```

**server.cfg beállítások:**
```cfg
set db_host "127.0.0.1"
set db_port 3306
set db_name "fivem_db"
set db_user "fivem_user"
set db_pass "secure_password_123"
```

### Spawn pozíció használata

```lua
-- Character spawn
local Locations = require('core.configs.sh_locations')

RegisterNetEvent('spawnCharacter', function()
    local source = source
    local character = GetCharacter(source)
    
    if not character then return end
    
    -- Mentett pozíció vagy alapértelmezett
    local coords = character:GetCoords() or Locations.Spawn
    
    -- Spawn
    TriggerClientEvent('setPlayerCoords', source, coords)
end)
```

### Loading screen pozíció

```lua
-- Karakter választás közben
local Locations = require('core.configs.sh_locations')

RegisterNetEvent('startCharacterSelection', function()
    local source = source
    
    -- Loading pozícióba helyezés
    TriggerClientEvent('setPlayerCoords', source, Locations.Loading)
    
    -- Karakter lista megnyitása
    TriggerClientEvent('openCharacterMenu', source)
end)
```

### Job beállítás config-ból

```lua
-- Job adás
local Groups = require('core.configs.sh_groups')

RegisterNetEvent('setPlayerJob', function(jobName, rank)
    local source = source
    local character = GetCharacter(source)
    
    if not character then return end
    
    -- Config ellenőrzés
    local jobConfig = Groups[jobName]
    if not jobConfig then
        return print('Nem létező job')
    end
    
    -- Rang ellenőrzés
    if not jobConfig.ranks[rank] then
        return print('Nem létező rang')
    end
    
    -- Job beállítás
    character:SetGroup(jobName, rank)
    
    print(('Job beállítva: %s - %s'):format(
        jobConfig.label,
        jobConfig.ranks[rank].label
    ))
end)
```

**Használat:**
```lua
local Locations = require('core.configs.sh_locations')

-- Spawn menü
local spawnMenu = {}
for id, spawn in pairs(Locations.SpawnPoints) do
    table.insert(spawnMenu, {
        label = spawn.label,
        coords = spawn.coords,
        heading = spawn.heading
    })
end

-- Spawn pont választása
RegisterNetEvent('selectSpawnPoint', function(spawnId)
    local source = source
    local character = GetCharacter(source)
    local spawn = Locations.SpawnPoints[spawnId]
    
    if spawn then
        TriggerClientEvent('spawnPlayer', source, spawn.coords, spawn.heading)
    end
end)
```

### Dynamic admin hierarchy

```lua
-- Admin jogok ellenőrzése
local AdminGroups = require('core.configs.sv_adminGroups')

local function HasPermission(playerGroup, requiredGroup)
    -- Saját csoport egyezik
    if playerGroup == requiredGroup then
        return true
    end
    
    -- Magasabb szintű csoport
    local currentGroup = playerGroup
    while currentGroup do
        local groupData = AdminGroups[currentGroup]
        if not groupData then break end
        
        if currentGroup == requiredGroup then
            return true
        end
        
        -- Következő magasabb csoport
        currentGroup = groupData.child and 
            GetParentGroup(currentGroup, AdminGroups)
    end
    
    return false
end

-- Segédfunkció parent megkeresésére
local function GetParentGroup(childGroup, groups)
    for groupName, groupData in pairs(groups) do
        if groupData.child == childGroup then
            return groupName
        end
    end
    return nil
end

-- Használat
local player = GetPlayer(source)
if HasPermission(player.group, 'admin') then
    print('Admin vagy magasabb szint')
end
```
